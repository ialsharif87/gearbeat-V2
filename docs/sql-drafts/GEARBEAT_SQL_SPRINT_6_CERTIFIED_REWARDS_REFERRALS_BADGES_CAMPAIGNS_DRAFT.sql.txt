-- ==========================================
-- 1. GEARBEAT CERTIFIED (Studios & Partners)
-- ==========================================
CREATE TABLE IF NOT EXISTS public.studio_tiers (
    tier_key TEXT PRIMARY KEY, -- verified, trusted, premium, elite, flagship
    name TEXT NOT NULL,
    description TEXT,
    min_audit_score INT DEFAULT 0,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.certified_studios (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    studio_id UUID REFERENCES public.studios(id) UNIQUE,
    tier_key TEXT REFERENCES public.studio_tiers(tier_key),
    status TEXT DEFAULT 'pending_audit', -- pending_audit, auditing, certified, suspended, expired
    last_audit_at TIMESTAMPTZ,
    audit_metadata JSONB DEFAULT '{}'::jsonb, -- Hardware audit, acoustic scores, etc.
    verification_token TEXT UNIQUE, -- For public QR verification
    valid_until DATE,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.studio_certification_history (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    certified_studio_id UUID REFERENCES public.certified_studios(id),
    old_status TEXT,
    new_status TEXT NOT NULL,
    actor_id UUID REFERENCES auth.users(id),
    notes TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ==========================================
-- 2. LOYALTY & REWARD TIERS
-- ==========================================
CREATE TABLE IF NOT EXISTS public.customer_tiers (
    tier_key TEXT PRIMARY KEY, -- standard, bronze, silver, gold, platinum, black
    name TEXT NOT NULL,
    min_points_ytd INT NOT NULL,
    benefits JSONB DEFAULT '[]'::jsonb,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.vendor_tiers (
    tier_key TEXT PRIMARY KEY, -- standard, pro, elite
    name TEXT NOT NULL,
    min_sales_ytd DECIMAL(12, 2),
    commission_discount DECIMAL(5, 2),
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.tier_rules (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    tier_type TEXT NOT NULL, -- customer, vendor, studio
    target_tier TEXT NOT NULL,
    metric_key TEXT NOT NULL, -- bookings_count, total_spend, review_avg
    threshold DECIMAL(12, 2) NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ==========================================
-- 3. REWARDS ENGINE
-- ==========================================
CREATE TABLE IF NOT EXISTS public.loyalty_reward_accounts (
    user_id UUID PRIMARY KEY REFERENCES auth.users(id),
    current_balance INT DEFAULT 0,
    lifetime_points INT DEFAULT 0,
    current_customer_tier TEXT REFERENCES public.customer_tiers(tier_key),
    last_milestone_at TIMESTAMPTZ,
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.reward_rules (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    action_key TEXT NOT NULL, -- booking_complete, order_paid, review_approved
    points_awarded INT NOT NULL,
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.reward_events (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID REFERENCES auth.users(id),
    rule_id UUID REFERENCES public.reward_rules(id),
    points_change INT NOT NULL, -- Positive for earn, negative for spend
    source_type TEXT NOT NULL, -- booking, order, review, manual
    source_id UUID,
    metadata JSONB,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),

    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),

    created_at TIMESTAMPTZ DEFAULT NOW()

CREATE TABLE IF NOT EXISTS public.campaign_codes (

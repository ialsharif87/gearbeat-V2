-- ==========================================
-- 1. PAYMENT INFRASTRUCTURE
-- ==========================================
CREATE TABLE IF NOT EXISTS public.payment_methods_reference (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    provider TEXT NOT NULL, -- manual, tap, stripe, etc.
    method_type TEXT NOT NULL, -- card, mada, bank_transfer, etc.
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Initial intent tracking
CREATE TABLE IF NOT EXISTS public.payment_sessions (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID REFERENCES auth.users(id),
    vertical_type TEXT NOT NULL, -- studio_booking, marketplace_order, etc.
    vertical_record_id UUID NOT NULL,
    amount DECIMAL(12, 2) NOT NULL,
    currency_code TEXT DEFAULT 'SAR',
    status TEXT DEFAULT 'created', -- created, redirected, abandoned, completed
    provider_session_id TEXT, -- e.g., tap_session_id
    metadata JSONB DEFAULT '{}'::jsonb,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Finalized transaction records
CREATE TABLE IF NOT EXISTS public.payment_transactions (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    session_id UUID REFERENCES public.payment_sessions(id),
    provider_transaction_id TEXT UNIQUE, -- e.g., tap_charge_id
    provider TEXT NOT NULL,
    amount DECIMAL(12, 2) NOT NULL,
    status TEXT NOT NULL, -- captured, failed, declined
    error_message TEXT,
    processed_at TIMESTAMPTZ,
    metadata JSONB DEFAULT '{}'::jsonb,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Pilot-specific manual bypass
CREATE TABLE IF NOT EXISTS public.manual_payment_confirmations (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    session_id UUID REFERENCES public.payment_sessions(id) UNIQUE,
    actor_id UUID REFERENCES auth.users(id), -- The admin who confirmed it
    confirmation_notes TEXT,
    proof_url TEXT, -- Link to bank receipt image
    confirmed_at TIMESTAMPTZ DEFAULT NOW()
);

-- ==========================================
-- 2. RELIABILITY & AUDIT
-- ==========================================
CREATE TABLE IF NOT EXISTS public.payment_webhook_events (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    provider TEXT NOT NULL,
    external_event_id TEXT NOT NULL,
    event_type TEXT NOT NULL,
    payload JSONB NOT NULL,
    status TEXT DEFAULT 'pending', -- pending, processed, failed
    processed_at TIMESTAMPTZ,
    UNIQUE(provider, external_event_id)
);

CREATE TABLE IF NOT EXISTS public.payment_idempotency_keys (
    idempotency_key TEXT PRIMARY KEY,
    user_id UUID REFERENCES auth.users(id),
    request_path TEXT,
    response_body JSONB,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Unified reference for vertical tables to join against
CREATE TABLE IF NOT EXISTS public.payment_references_unified (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    vertical_type TEXT NOT NULL,
    vertical_record_id UUID NOT NULL,
    transaction_id UUID REFERENCES public.payment_transactions(id),
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ==========================================
-- 3. REFUNDS
-- ==========================================
CREATE TABLE IF NOT EXISTS public.refund_requests (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    transaction_id UUID REFERENCES public.payment_transactions(id),
    requested_by UUID REFERENCES auth.users(id),
    amount DECIMAL(12, 2) NOT NULL,
    reason TEXT,
    status TEXT DEFAULT 'pending', -- pending, approved, rejected, processed
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.refund_transactions (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    refund_request_id UUID REFERENCES public.refund_requests(id),
    provider_refund_id TEXT UNIQUE,
    amount DECIMAL(12, 2) NOT NULL,
    status TEXT NOT NULL,
    processed_at TIMESTAMPTZ,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ==========================================
-- 4. PAYOUTS & COMMISSIONS
-- ==========================================
CREATE TABLE IF NOT EXISTS public.payout_accounts (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    partner_account_id UUID REFERENCES public.partner_accounts(id),
    provider TEXT DEFAULT 'bank_transfer',
    account_details JSONB, -- Encrypted or vault-stored in production
    status TEXT DEFAULT 'pending_verification',
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.payout_batches (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    batch_number TEXT UNIQUE NOT NULL,
    status TEXT DEFAULT 'draft', -- draft, scheduled, processing, completed
    total_amount DECIMAL(12, 2) NOT NULL,
    scheduled_for TIMESTAMPTZ,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.payout_items (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    batch_id UUID REFERENCES public.payout_batches(id),
    partner_account_id UUID REFERENCES public.partner_accounts(id),
    payout_account_id UUID REFERENCES public.payout_accounts(id),
    amount DECIMAL(12, 2) NOT NULL,
    status TEXT DEFAULT 'pending',
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.commission_rules (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    vertical_type TEXT, -- NULL for global default
    partner_account_id UUID, -- NULL for global default
    commission_rate DECIMAL(5, 2) NOT NULL, -- e.g., 15.00
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.commission_calculations (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    transaction_id UUID REFERENCES public.payment_transactions(id),
    gross_amount DECIMAL(12, 2) NOT NULL,
    commission_rate DECIMAL(5, 2) NOT NULL,
    commission_amount DECIMAL(12, 2) NOT NULL,
    net_payout_amount DECIMAL(12, 2) NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ==========================================
-- 5. FINANCE LEDGER & RECONCILIATION
-- ==========================================
CREATE TABLE IF NOT EXISTS public.finance_ledger_entries (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    partner_account_id UUID REFERENCES public.partner_accounts(id),
    entry_type TEXT NOT NULL, -- credit, debit
    amount DECIMAL(12, 2) NOT NULL,
    description TEXT,
    reference_type TEXT, -- payment, refund, payout, commission
    reference_id UUID,
    metadata JSONB,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.reconciliation_runs (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    run_date DATE DEFAULT CURRENT_DATE,
    provider TEXT NOT NULL,
    status TEXT DEFAULT 'in_progress', -- in_progress, completed, flagged
    discrepancy_count INT DEFAULT 0,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.reconciliation_items (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    run_id UUID REFERENCES public.reconciliation_runs(id),
    provider_reference_id TEXT,
    internal_transaction_id UUID REFERENCES public.payment_transactions(id),
    amount_discrepancy DECIMAL(12, 2),
    status_discrepancy TEXT,
    notes TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Billing references
CREATE TABLE IF NOT EXISTS public.invoice_references (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    vertical_type TEXT NOT NULL,
    vertical_record_id UUID NOT NULL,
    invoice_number TEXT UNIQUE NOT NULL,
    invoice_url TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

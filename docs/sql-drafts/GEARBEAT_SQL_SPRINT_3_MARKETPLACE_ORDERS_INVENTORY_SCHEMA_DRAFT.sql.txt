-- ==========================================
-- 1. VENDOR & CATEGORY CORE
-- ==========================================
CREATE TABLE IF NOT EXISTS public.marketplace_vendors (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    partner_account_id UUID REFERENCES public.partner_accounts(id),
    business_name TEXT NOT NULL,
    support_email TEXT,
    commission_rate_override DECIMAL(5, 2), -- e.g., 12.50 for 12.5%
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.vendor_profiles (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    vendor_id UUID REFERENCES public.marketplace_vendors(id) ON DELETE CASCADE,
    slug TEXT UNIQUE NOT NULL,
    store_name TEXT NOT NULL,
    store_description TEXT,
    banner_url TEXT,
    logo_url TEXT,
    metadata JSONB DEFAULT '{}'::jsonb,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.product_categories (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    parent_id UUID REFERENCES public.product_categories(id),
    name TEXT NOT NULL,
    slug TEXT UNIQUE NOT NULL,
    icon_url TEXT,
    sort_order INT DEFAULT 0,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ==========================================
-- 2. PRODUCT CATALOG
-- ==========================================
CREATE TABLE IF NOT EXISTS public.marketplace_products (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    vendor_id UUID REFERENCES public.marketplace_vendors(id),
    category_id UUID REFERENCES public.product_categories(id),
    name TEXT NOT NULL,
    slug TEXT UNIQUE NOT NULL,
    description TEXT,
    brand_name TEXT,
    base_price DECIMAL(12, 2) NOT NULL,
    sale_price DECIMAL(12, 2),
    currency_code TEXT DEFAULT 'SAR',
    status TEXT DEFAULT 'draft', -- draft, approved, active, out_of_stock, archived
    is_active BOOLEAN DEFAULT true,
    metadata JSONB DEFAULT '{}'::jsonb,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.product_media (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    product_id UUID REFERENCES public.marketplace_products(id) ON DELETE CASCADE,
    url TEXT NOT NULL,
    media_type TEXT DEFAULT 'image', -- image, video
    sort_order INT DEFAULT 0,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Variations: e.g., Professional vs Home version, colors, etc.
CREATE TABLE IF NOT EXISTS public.product_variants (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    product_id UUID REFERENCES public.marketplace_products(id) ON DELETE CASCADE,
    sku TEXT UNIQUE NOT NULL,
    name TEXT, -- e.g., "Silver", "1TB Version"
    price_override DECIMAL(12, 2),
    weight_kg DECIMAL(8, 2),
    dimensions JSONB, -- {length, width, height}
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ==========================================
-- 3. INVENTORY & RESERVATIONS
-- ==========================================
CREATE TABLE IF NOT EXISTS public.product_inventory (
    variant_id UUID PRIMARY KEY REFERENCES public.product_variants(id) ON DELETE CASCADE,
    quantity_available INT NOT NULL DEFAULT 0 CHECK (quantity_available >= 0),
    quantity_reserved INT NOT NULL DEFAULT 0 CHECK (quantity_reserved >= 0),
    low_stock_threshold INT DEFAULT 5,
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Temporary hold during checkout (e.g., 15 mins)
CREATE TABLE IF NOT EXISTS public.inventory_reservations (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    variant_id UUID REFERENCES public.product_variants(id),
    user_id UUID REFERENCES auth.users(id),
    quantity INT NOT NULL,
    expires_at TIMESTAMPTZ NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ==========================================
-- 4. CARTS & ORDERS
-- ==========================================
CREATE TABLE IF NOT EXISTS public.carts (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID REFERENCES auth.users(id) UNIQUE,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.cart_items (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    cart_id UUID REFERENCES public.carts(id) ON DELETE CASCADE,
    variant_id UUID REFERENCES public.product_variants(id),
    quantity INT NOT NULL CHECK (quantity > 0),
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.marketplace_orders (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    user_id UUID REFERENCES auth.users(id),
    order_number TEXT UNIQUE NOT NULL, -- e.g., GB-ORD-12345
    status TEXT DEFAULT 'pending_payment', -- pending_payment, paid, processing, shipped, delivered, cancelled, refunded
    total_amount DECIMAL(12, 2) NOT NULL,
    currency_code TEXT DEFAULT 'SAR',
    shipping_address JSONB NOT NULL,
    billing_address JSONB,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.marketplace_order_items (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    order_id UUID REFERENCES public.marketplace_orders(id) ON DELETE CASCADE,
    variant_id UUID REFERENCES public.product_variants(id),
    vendor_id UUID REFERENCES public.marketplace_vendors(id),
    quantity INT NOT NULL,
    price_at_time DECIMAL(12, 2) NOT NULL,
    fulfillment_status TEXT DEFAULT 'pending', -- pending, packed, shipped, delivered, cancelled
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.order_status_history (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    order_id UUID REFERENCES public.marketplace_orders(id),
    status TEXT NOT NULL,
    actor_id UUID REFERENCES auth.users(id),
    notes TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ==========================================
-- 5. FULFILLMENT & POST-PURCHASE
-- ==========================================
CREATE TABLE IF NOT EXISTS public.fulfillment_shipments (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    order_id UUID REFERENCES public.marketplace_orders(id),
    vendor_id UUID REFERENCES public.marketplace_vendors(id),
    carrier TEXT, -- Aramex, SMSA, etc.
    tracking_number TEXT,
    shipping_label_url TEXT,
    estimated_delivery DATE,
    actual_delivery TIMESTAMPTZ,
    status TEXT DEFAULT 'preparing',
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.return_requests (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    order_item_id UUID REFERENCES public.marketplace_order_items(id),
    reason TEXT NOT NULL,
    status TEXT DEFAULT 'pending', -- pending, approved, rejected, completed
    images JSONB, -- Photos of defect
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.refund_requests (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    order_id UUID REFERENCES public.marketplace_orders(id),
    amount DECIMAL(12, 2) NOT NULL,
    reason TEXT,
    status TEXT DEFAULT 'pending',
    external_refund_reference TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ==========================================
-- 6. PAYMENT & PAYOUT REFERENCES
-- ==========================================
CREATE TABLE IF NOT EXISTS public.marketplace_payment_references (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    order_id UUID REFERENCES public.marketplace_orders(id),
    provider TEXT DEFAULT 'manual',
    reference_id TEXT,
    amount DECIMAL(12, 2) NOT NULL,
    status TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS public.vendor_payout_references (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    order_id UUID REFERENCES public.marketplace_orders(id),
    vendor_id UUID REFERENCES public.marketplace_vendors(id),
    payout_amount DECIMAL(12, 2) NOT NULL, -- total minus commission
    status TEXT DEFAULT 'pending', -- pending, scheduled, paid, held
    payout_date TIMESTAMPTZ,
    created_at TIMESTAMPTZ DEFAULT NOW()
);
